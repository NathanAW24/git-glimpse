\n- suguru: workaround for the issue is putting `.pnpmfile.cjs` in my repository.

```js
module.exports = {
  hooks: {
    readPackage
  }
}
 
function readPackage (pkg, ctx) {
  if (pkg.name === "@nextui-org/react") {
    pkg.dependencies = {
      ...pkg.dependencies,
      "@react-stately/collections": "3.4.2",
      "@react-stately/menu": "3.4.0",
    }
  }
  return pkg
}
